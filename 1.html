spread operator

spreads any iterables, so converting iterables to arrays is easy 

const map = new Map([["thunder", "lightning"],["fire","ice"]]);
const keys = [...map.keys()]
console.log(keys)
const values = [...map.values()]
console.log(values)

// list of unique entries by using new Set()
const entries = ["apple", "banana", "grape", "banana"]
console.log([...new Set(entries)])

// array of DOM elements
const errorMessageNodes = [...document.getElementsByClassName("error")]
console.log(errorMessageNodes)

// characters in a string
console.log([..."hello"])


concatenate arrays

    const lat = ["N", "S"]
    const long = ["E", "W"]
    const directions = [...lat, ...long]
    console.log(directions)
    // -> [ 'N', 'S', 'E', 'W' ]


copy arrays

   const source = [{"number" : 1}, {"number" : 2}, {"number" : 3}]
    const copy = [...source]
    console.log(copy)
    // -> [ 1, 2, 3 ]

    source[0].number += 1 // also changes entry of copy
    console.log(copy)

    source[0] = undefined // but changing one array doesn't change the other
    console.log(source)
    console.log(copy)
